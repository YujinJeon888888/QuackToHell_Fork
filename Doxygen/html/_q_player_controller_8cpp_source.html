37void AQPlayerController::SetupInputComponent()
38{
39 Super::SetupInputComponent();
40
41 //InputComponent는 Actor의 멤버변수이다. 멤버변수를 EnhancedInput으로 캐스팅한다.
42 auto* EnhancedInputComponent = Cast<UEnhancedInputComponent>(InputComponent);
43 if (EnhancedInputComponent) {
44 /*액션에 대해, 트리거 되면 콜백되는 함수를 바인딩한다.*/
45 EnhancedInputComponent->BindAction(MoveAction, ETriggerEvent::Triggered, this, &ThisClass::InputMove);
46 EnhancedInputComponent->BindAction(TurnAction, ETriggerEvent::Triggered, this, &ThisClass::InputTurn);
47 EnhancedInputComponent->BindAction(InteractionAction, ETriggerEvent::Triggered, this, &ThisClass::InputInteraction);
48 EnhancedInputComponent->BindAction(ConversingQuitAction, ETriggerEvent::Triggered, this, &ThisClass::InputConversingQuit);
49 EnhancedInputComponent->BindAction(EnableTurnAction, ETriggerEvent::Started, this, &ThisClass::InputEnableTurn);
50 EnhancedInputComponent->BindAction(EnableTurnAction, ETriggerEvent::Completed, this, &ThisClass::InputEnableTurn);
51 EnhancedInputComponent->BindAction(InteractionAction, ETriggerEvent::Started, this, &ThisClass::InputEnableInteracton);
52 EnhancedInputComponent->BindAction(InteractionAction, ETriggerEvent::Completed, this, &ThisClass::InputEnableInteracton);
53 EnhancedInputComponent->BindAction(ConversingQuitAction, ETriggerEvent::Started, this, &ThisClass::InputEnableConversingQuit);
54 EnhancedInputComponent->BindAction(ConversingQuitAction, ETriggerEvent::Completed, this, &ThisClass::InputEnableConversingQuit);
55
56
57 }

82void AQPlayerController::StartDialog()
83{
84
85 //0. 상대방 NPC를 불러옴
86 TObjectPtr<AQPlayer> _Player = Cast<AQPlayer>(this->GetPawn());
87 TObjectPtr<AQNPC> NPC = Cast<AQNPC>(_Player->GetClosestNPC());
88 //1. 대화 가능한지 check
89 //임시 지우고 주석풀기 : if (Cast<AQPlayer>(this->GetPawn())->GetCanStartConversP2N(NPC))
90 {
91 //2. 상대방 NPC의 컨트롤러를 불러옴
92 TObjectPtr<AQNPCController> NPCController = Cast<AQNPCController>(NPC->GetController());
93
94 //3. 대화 시작하라고 명령한다.
95 NPCController->StartDialog();
96
97 //4. 플레이어를 대화처리한다.
98 this->ConverseProcess(NPC);
99
100 //5. P2N Widget에게 자신의 정보를 넘긴다.
101 //내 정보 넘겨주기
102 Cast<UQP2NWidget>((VillageUIManager->GetActiveWidgets())[EVillageUIType::P2N])->SetConversingPlayer(this);
103
104 }

185void AQPlayerController::InputMove(const FInputActionValue& InputValue)
186{
187
188 FVector2D MovementVector = InputValue.Get<FVector2D>();
189
190 //앞 이동
191 if (MovementVector.X != 0) {
192 /*내가 바라보는 곳 기준의 앞 방향을 구해줌*/
193 //마우스 방향 회전값 기준
194 FRotator Rotator = this->GetControlRotation();
195 //해당 회전값의 기준의 앞 방향을 구해준다.
196 FVector Direction = UKismetMathLibrary::GetForwardVector(FRotator(0, Rotator.Yaw, 0));
197 //방향과 크기를 전달함. (시간마다 거리를 더하는 것은 MovementComponent에서, 속도 세팅은 movementcomponent에서)
198 this->GetPawn()->AddMovementInput(Direction, MovementVector.X);
199 }
200 //옆 이동
201 if (MovementVector.Y != 0) {
202 /*내가 바라보는 곳 기준의 옆 방향을 구해줌*/
203 //마우스 방향 회전값 기준
204 FRotator Rotator = this->GetControlRotation();
205 //해당 회전값의 기준의 옆 방향을 구해준다.
206 FVector Direction = UKismetMathLibrary::GetRightVector(FRotator(0, Rotator.Yaw, 0));
207 //방향과 크기를 전달함. (시간마다 거리를 더하는 것은 MovementComponent에서, 속도 세팅은 movementcomponent에서)
208 this->GetPawn()->AddMovementInput(Direction, MovementVector.Y);
209 }
210}
